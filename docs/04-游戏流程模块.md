# 游戏流程模块

本文档详细描述游戏流程控制相关的功能实现。

## 一、模块概述

游戏流程模块负责控制游戏的进行，包括发牌、摸牌、出牌、回合流转等核心流程。

## 三、核心函数

### 3.1 游戏开始

#### startGame()

**功能**: 开始新一局游戏

**代码位置**: server.js:713-832

**处理流程**:

```javascript
startGame() {
    if (this.gameRunning) return;
    
    // 1. 增加局数
    this.currentRound++;
    if (!this.matchStarted) {
        this.matchStarted = true;
        this.matchScores = [0, 0, 0, 0];
        this.roundHistory = [];
    }
    
    this.gameRunning = true;
    
    // 2. 荒番局逻辑
    this.isHuangFanRound = this.huangFanCount > 0;
    
    // 3. 创建并洗牌
    let deck;
    if (this.isTestRoom && this.customDeckOrder) {
        deck = createDeckWithOrder(this.customDeckOrder);
    } else {
        deck = shuffleDeck(createDeck());
    }
    
    // 4. 确定庄家
    let dealerIndex;
    if (this.currentRound === 1) {
        dealerIndex = Math.floor(Math.random() * 4);  // 第一局随机
    } else if (this.lastWinnerIndex >= 0) {
        dealerIndex = this.lastWinnerIndex;  // 上一局赢家做庄
    } else {
        dealerIndex = Math.floor(Math.random() * 4);
    }
    
    // 5. 初始化游戏状态
    this.gameState = {
        deck: deck,
        dealerIndex: dealerIndex,
        currentPlayerIndex: dealerIndex,
        turnPhase: 'draw',
        lastDiscard: null,
        lastDiscardPlayer: -1,
        pendingActions: [],
        actionTimeout: null,
        discardTimeout: null,
        lastDrawnTile: null,
        roundNumber: 1,
        gameOver: false,
        waitingForQiao: false,
        huangFanRound: this.huangFanCount > 0,
        huangFanCount: this.huangFanCount,
        cangyingTile: null,
        gangShangPao: false
    };
    
    // 6. 发牌
    this.players.forEach((player, index) => {
        // 重置玩家状态
        player.hand = [];
        player.melds = [];
        player.discards = [];
        player.flowers = [];
        player.isTing = false;
        player.isQiao = false;
        player.sankouCounts = [0, 0, 0, 0];
        
        // 根据庄家位置动态计算风牌
        player.wind = WINDS[(index - dealerIndex + 4) % 4];
        
        // 发牌（庄家14张，其他13张）
        const cardCount = index === dealerIndex ? 14 : 13;
        for (let i = 0; i < cardCount; i++) {
            this.drawTileForPlayer(player, true);  // 发牌阶段
        }
        player.hand = sortTiles(player.hand);
    });
    
    // 7. 广播游戏开始
    this.broadcastGameStart();
    
    // 8. 检查庄家是否有暗杠
    const dealer = this.players[dealerIndex];
    const dealerAnGangActions = this.checkAnGangAvailable(dealer);
    
    if (dealerAnGangActions.length > 0) {
        // 庄家有暗杠，设置待处理状态
        this.gameState.turnPhase = 'discard';
        this.gameState.pendingAnGang = {
            playerId: dealer.id,
            playerIndex: dealer.seatIndex,
            anGangOptions: dealerAnGangActions
        };
        // ... 通知庄家
    } else {
        // 没有暗杠，直接进入出牌阶段
        this.gameState.turnPhase = 'discard';
        this.notifyCurrentPlayer();
    }
}
```

### 3.2 摸牌流程

#### playerDraw(socketId)

**功能**: 玩家摸牌

**代码位置**: server.js:1188-1320

**参数**:
| 参数 | 类型 | 说明 |
|------|------|------|
| socketId | string | 玩家Socket ID |

**返回值**: `{success: boolean, tile?: Tile, error?: string}`

**处理流程**:

```javascript
playerDraw(socketId) {
    const player = this.players.find(p => p.id === socketId);
    if (!player) return;
    
    // 1. 验证回合
    if (this.gameState.currentPlayerIndex !== player.seatIndex) {
        return { error: '不是你的回合' };
    }
    if (this.gameState.turnPhase !== 'draw') {
        return { error: '当前不能摸牌' };
    }
    if (this.gameState.deck.length === 0) {
        this.endRound('draw', -1, -1, false, false);
        return;
    }
    
    // 2. 摸牌（包含补花）
    const tile = this.drawTileForPlayer(player, false);
    if (!tile) {
        this.endRound('draw', -1, -1, false, false);
        return;
    }
    
    // 3. 记录摸的牌（用于超时自动出牌）
    this.gameState.lastDrawnTile = tile;
    
    // 4. 进入出牌阶段
    this.gameState.turnPhase = 'discard';
    
    // 5. 检查自摸（只有敲牌后才能自摸）
    if (player.isQiao && this.canHu(player.hand, player.melds)) {
        this.gameState.pendingZimo = {
            playerId: player.id,
            playerIndex: player.seatIndex,
            tile: tile
        };
        player.socket.emit('action_available', {
            actions: ['hu_zimo'],
            tile: tile
        });
    }
    
    // 6. 检查暗杠
    const anGangActions = this.checkAnGangAvailable(player);
    if (anGangActions.length > 0 && !this.gameState.pendingZimo) {
        this.gameState.pendingAnGang = {
            playerId: player.id,
            playerIndex: player.seatIndex,
            anGangOptions: anGangActions
        };
        player.socket.emit('action_available', {
            actions: ['an_gang'],
            tile: anGangActions[0].tile,
            anGangOptions: anGangActions
        });
    }
    
    // 7. 检查加杠
    // ... 类似暗杠处理
    
    // 8. 广播状态
    this.broadcastGameState();
    
    // 9. 通知玩家摸到的牌
    player.socket.emit('tile_drawn', { tile: tile });
    
    // 10. 设置出牌超时
    if (!player.isBot && !player.offline) {
        this.setDiscardTimeout(player);
    }
    
    return { success: true, tile: tile };
}
```

**功能**: 为玩家摸一张牌（处理花牌补花）

**参数**:
| 参数 | 类型 | 说明 |
|------|------|------|
| player | Player | 玩家对象 |
| isDealingPhase | boolean | 是否发牌阶段 |

**返回值**: Tile | null

**处理逻辑**:

```javascript
drawTileForPlayer(player, isDealingPhase = false) {
    if (this.gameState.deck.length === 0) {
        return null;
    }
    
    let tile = this.gameState.deck.pop();
    
    // 如果是花牌，放入花牌区并继续摸
    while (isFlowerTile(tile)) {
        player.flowers.push(tile);
        
        // 游戏中广播补花事件
        if (!isDealingPhase) {
            this.broadcast('flower_drawn', {
                playerIndex: player.seatIndex,
                playerId: player.id,
                flower: tile,
                flowerName: getFlowerName(tile),
                totalFlowers: player.flowers.length
            });
        }
        
        if (this.gameState.deck.length === 0) {
            return null;
        }
        tile = this.gameState.deck.pop();
    }
    
    player.hand.push(tile);
    return tile;
}
```

### 3.3 出牌流程

#### playerDiscard(socketId, tileId)

**功能**: 玩家出牌

**代码位置**: server.js:1323-1437

**参数**:
| 参数 | 类型 | 说明 |
|------|------|------|
| socketId | string | 玩家Socket ID |
| tileId | string | 要出的牌ID |

**返回值**: `{success?: boolean, error?: string}`

**处理流程**:

```javascript
playerDiscard(socketId, tileId) {
    const player = this.players.find(p => p.id === socketId);
    if (!player) return { error: '玩家不存在' };
    
    // 1. 验证回合
    if (this.gameState.currentPlayerIndex !== player.seatIndex) {
        return { error: '不是你的回合' };
    }
    if (this.gameState.turnPhase !== 'discard') {
        return { error: '当前不能出牌' };
    }
    
    // 2. 敲牌限制：只能打刚摸的牌
    if (player.isQiao && this.gameState.lastDrawnTile) {
        if (tileId !== this.gameState.lastDrawnTile.id) {
            return { error: '已敲牌，只能打刚摸的牌！' };
        }
    }
    
    // 3. 清除出牌超时
    if (this.gameState.discardTimeout) {
        clearTimeout(this.gameState.discardTimeout);
        this.gameState.discardTimeout = null;
    }
    
    // 4. 执行出牌
    const tileIndex = player.hand.findIndex(t => t.id === tileId);
    if (tileIndex === -1) {
        return { error: '没有这张牌' };
    }
    
    const tile = player.hand.splice(tileIndex, 1)[0];
    player.discards.push(tile);
    player.hand = sortTiles(player.hand);
    
    // 5. 更新游戏状态
    this.gameState.lastDiscard = tile;
    this.gameState.lastDiscardPlayer = player.seatIndex;
    this.gameState.lastDrawnTile = null;
    this.gameState.gangShangPao = false;
    
    // 6. 检查听牌敲牌
    const tingTiles = this.getTingTiles(player.hand, player.melds);
    if (!player.isTing && !player.isQiao && tingTiles.length > 0) {
        player.isTing = true;
        this.gameState.waitingForQiao = true;
        
        // 设置敲牌确认超时（30秒）
        this.gameState.qiaoTimeout = setTimeout(() => {
            // 超时自动敲牌
            player.isQiao = true;
            this.gameState.waitingForQiao = false;
            // ... 继续游戏流程
        }, 30000);
        
        // 通知玩家可以敲牌
        player.socket.emit('ting_and_qiao_prompt', {
            message: '🎯 您已听牌！是否敲牌？',
            tingTiles: tingTiles
        });
        
        // 广播出牌
        this.broadcast('tile_discarded', {
            playerIndex: player.seatIndex,
            tile: tile,
            tileName: getTileName(tile)
        });
        
        return { success: true };
    }
    
    // 7. 广播出牌
    this.broadcast('tile_discarded', {
        playerIndex: player.seatIndex,
        tile: tile,
        tileName: getTileName(tile)
    });
    
    // 8. 检查其他玩家动作
    this.checkActionsAfterDiscard(tile, player.seatIndex);
    
    return { success: true };
}
```

notifyCurrentPlayer() {
    // 如果游戏暂停，停止通知
    if (this.isPaused) return;
    
    const currentPlayer = this.players[this.gameState.currentPlayerIndex];
    
    // 清除之前的出牌超时
    if (this.gameState.discardTimeout) {
        clearTimeout(this.gameState.discardTimeout);
        this.gameState.discardTimeout = null;
    }
    
    // 检查待处理的暗杠
    if (this.gameState.pendingAnGang && 
        this.gameState.pendingAnGang.playerIndex === currentPlayer.seatIndex) {
        if (currentPlayer.isBot || currentPlayer.offline || currentPlayer.aiTakeover) {
            setTimeout(() => this.aiHandleAnGang(currentPlayer), 500);
        } else {
            this.broadcastGameState();
            this.setDiscardTimeout(currentPlayer);
        }
        return;
    }
    
    // 检查待处理的加杠
    if (this.gameState.pendingJiaGang && 
        this.gameState.pendingJiaGang.playerIndex === currentPlayer.seatIndex) {
        // ... 类似暗杠处理
        return;
    }
    
    // 根据玩家类型处理
    if (currentPlayer.isBot) {
        // AI玩家自动行动
        setTimeout(() => this.aiAction(currentPlayer), 
            500 + Math.random() * 500);
    } else if (currentPlayer.offline || currentPlayer.aiTakeover) {
        // 离线或被接管的玩家当作AI处理
        setTimeout(() => this.aiAction(currentPlayer), 500);
    } else {
        // 真人玩家
        if (this.gameState.turnPhase === 'discard') {
            this.setDiscardTimeout(currentPlayer);
        }
        this.broadcastGameState();
    }
}

#### autoDiscard(player)

**功能**: 超时自动出牌

**代码位置**: server.js:1018-1083

```javascript
autoDiscard(player) {
    if (!this.gameRunning) return;
    if (player.hand.length === 0) {
        if (this.gameState.deck.length === 0) {
            this.endRound('draw', -1, -1, false, false);
        }
        return;
    }
    
    // 优先打出刚摸的牌
    let tileToDiscard = this.gameState.lastDrawnTile;
    
    // 检查这张牌是否还在手牌中
    if (tileToDiscard) {
        const stillInHand = player.hand.find(t => t.id === tileToDiscard.id);
        if (!stillInHand) {
            tileToDiscard = null;
        }
    }
    
    // 如果没有记录或已不在手牌，打最后一张
    if (!tileToDiscard) {
        tileToDiscard = player.hand[player.hand.length - 1];
    }
    
    // 执行出牌
    const tileIndex = player.hand.findIndex(t => t.id === tileToDiscard.id);
    const tile = player.hand.splice(tileIndex, 1)[0];
    player.discards.push(tile);
    player.hand = sortTiles(player.hand);
    
    this.gameState.lastDiscard = tile;
    this.gameState.lastDiscardPlayer = player.seatIndex;
    this.gameState.lastDrawnTile = null;
    
    // 广播超时自动出牌
    this.broadcast('tile_discarded', {
        playerIndex: player.seatIndex,
        tile: tile,
        tileName: getTileName(tile),
        isAutoDiscard: true
    });
    
    // 通知该玩家
    if (player.socket) {
        player.socket.emit('auto_discard', { 
            tile: tile,
            message: '出牌超时，已自动打出' 
        });
    }
    
    // 检查其他玩家动作
    this.checkActionsAfterDiscard(tile, player.seatIndex);
}
```

endRound(resultType, winnerIndex = -1, loserIndex = -1, isZimo = false, isGangKai = false, isGangShangPao = false) {
    this.gameRunning = false;
    this.gameState.gameOver = true;
    
    // 清除所有超时计时器
    if (this.gameState.actionTimeout) clearTimeout(this.gameState.actionTimeout);
    if (this.gameState.discardTimeout) clearTimeout(this.gameState.discardTimeout);
    
    // 处理荒番逻辑
    const isDraw = (resultType === 'draw');
    if (isDraw) {
        this.huangFanCount++;
    }
    
    let roundResult = {
        round: this.currentRound,
        resultType: resultType,
        isZimo: isZimo,
        isGangShangPao: isGangShangPao,
        winnerIndex: winnerIndex,
        loserIndex: loserIndex,
        scoreResult: null,
        players: [],
        huangFanCount: this.huangFanCount,
        isHuangFanRound: this.isHuangFanRound
    };
    
    // 如果有人胡牌，计算积分
    if (winnerIndex >= 0) {
        const winner = this.players[winnerIndex];
        
        // 飞苍蝇
        if (this.gameState.deck && this.gameState.deck.length > 0) {
            this.gameState.cangyingTile = this.gameState.deck.pop();
        }
        
        // 计算番、花、苍蝇、分数
        const fanResult = this.calculateFan(winner, isZimo, isGangKai);
        const huaResult = this.calculateHua(winner);
        const cangyingResult = this.calculateCangying(winner, this.gameState.cangyingTile, fanResult);
        const scoreResult = this.calculateScore(winner, loserIndex, fanResult, huaResult, cangyingResult, isZimo, this.isHuangFanRound, this.huangFanCount, isGangShangPao);
        
        // 更新累计积分
        for (let i = 0; i < 4; i++) {
            this.matchScores[i] += scoreResult.scoreChanges[i];
        }
        
        roundResult.scoreResult = scoreResult;
        this.lastWinnerIndex = winnerIndex;
    }
    
    // 记录玩家信息
    roundResult.players = this.players.map((p, idx) => ({
        username: p.username,
        seatIndex: p.seatIndex,
        hand: p.hand,
        melds: p.melds,
        roundScore: roundResult.scoreResult ? roundResult.scoreResult.scoreChanges[idx] : 0,
        totalScore: this.matchScores[idx]
    }));
    
    // 保存历史记录
    this.roundHistory.push(roundResult);
    
    // 判断是否结束比赛
    if (this.currentRound >= this.totalRounds) {
        this.endMatch();
    } else {
        // 重置准备状态
        this.players.forEach(p => {
            p.ready = false;
            if (!p.isBot && !p.offline) {
                p.aiTakeover = false;
            }
        });
        
        // 广播本局结束
        this.broadcast('round_ended', {
            roundResult: roundResult,
            currentRound: this.currentRound,
            totalRounds: this.totalRounds,
            matchScores: this.matchScores,
            countdownSeconds: 30
        });
        
        // 启动倒计时
        this.startNextRoundCountdown();
    }
}

broadcastGameState(forceFullUpdate = false) {
    const now = Date.now();
    
    // 节流：100ms内只发送一次
    if (!forceFullUpdate && this._lastBroadcast && now - this._lastBroadcast < 100) {
        if (this._pendingBroadcast) return;
        this._pendingBroadcast = setTimeout(() => {
            this._pendingBroadcast = null;
            this.broadcastGameState(false);
        }, 100);
        return;
    }
    
    this._lastBroadcast = now;
    
    // 广播给玩家
    this.players.forEach(player => {
        if (player.socket) {
            player.socket.emit('game_state_update', {
                gameState: this.getPlayerGameState(player.id)
            });
        }
    });
    
    // 广播给观战者
    this.viewers.forEach(viewer => {
        if (viewer.socket) {
            viewer.socket.emit('game_state_update', {
                gameState: this.getViewerGameState()
            });
        }
    });
}
